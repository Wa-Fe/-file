    # js代码

        // 初始化设置透明程度 
    opacity: 0,

        // vue方法
        mounted: function() {
            window.addEventListener('scroll', this.handleScroll)
        },

        unmounted: function() {
            window.removeEventListener('scroll', this.handleScroll)
        },

        methods: {
            handleScroll: function() {
                var top = document.documentElement.scrollTop
                this.opacity = top / 300;
                if (this.opacity > 0.1) {
                    vm.title = "商品详情";
                } else {
                    vm.title = "";
                }
            },
        }
