        // data
    animate: false,

        // 消息滚动
        noticeScroll() {
            // setInterval('AutoScroll(".rolltext")', 3000)
            setInterval(() => {
                AutoScroll(".rolltext")
            }, 3000)
        },

        function AutoScroll(obj) {
            if (vm.message.length !== 0) {
                vm.animate = true;
                setTimeout(() => {
                    vm.message.push(vm.message[0]);
                    vm.message.shift();
                    vm.animate = false;
                }, 1000)
            }

        }
